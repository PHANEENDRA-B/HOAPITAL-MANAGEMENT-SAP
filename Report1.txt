FORM display_patients.


SELECT * FROM ZBB_PATIENT INTO TABLE GT_PATIENT.

IF S_PID IS NOT INITIAL.
  DELETE GT_PATIENT WHERE PATIENT_ID NOT IN S_PID.
ENDIF.


IF P_DEPT IS NOT INITIAL.
  DELETE GT_PATIENT WHERE DEPARTMENT_CODE <> P_DEPT.
ENDIF.

IF P_YEAR IS NOT INITIAL.
  DELETE gt_patient WHERE registration_date(4) <> p_year.
ENDIF.

  IF gt_patient IS INITIAL.
    MESSAGE 'No records found' TYPE 'I'.
    EXIT.
  ENDIF.


  ls_list-typ  = 'H'.
  ls_list-info = 'Patient Report'.
  APPEND ls_list TO lt_list.

  CLEAR ls_list.
  ls_list-typ  = 'S'.
  ls_list-key  = 'Run Date'.
  ls_list-info = sy-datum.
  APPEND ls_list TO lt_list.

  CLEAR ls_list.
  ls_list-typ  = 'S'.
  ls_list-key  = 'Run Time'.
  ls_list-info = sy-uzeit.
  APPEND ls_list TO lt_list.

  CLEAR ls_list.
  ls_list-typ  = 'S'.
  ls_list-key  = 'Run By'.
  ls_list-info = sy-uname.
  APPEND ls_list TO lt_list.



gs_layout-zebra = 'X'.                  " alternate row colors
gs_layout-colwidth_optimize = 'X'.

  CALL FUNCTION 'REUSE_ALV_GRID_DISPLAY'
    EXPORTING
      i_callback_program     = sy-repid
      i_structure_name ='ZBB_PATIENT'
      is_layout   = gs_layout
      i_callback_top_of_page = 'TOP_OF_PAGE'
    TABLES
      t_outtab    = gt_patient.

ENDFORM.

FORM register_patient.

  DATA: lv_last TYPE zbb_patient-patient_id.

  SELECT MAX( patient_id )
    INTO lv_last
    FROM zbb_patient.

  gv_new_id_num = lv_last.
  gv_new_id_num = gv_new_id_num + 1.
  gv_new_id = gv_new_id_num.


  CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
    EXPORTING
      input  = gv_new_id
    IMPORTING
      output = gv_new_id.

  CALL FUNCTION 'ENQUEUE_EZBB_PATIENT'
    EXPORTING
      patient_id = gv_new_id
    EXCEPTIONS
      foreign_lock = 1
      system_failure = 2
      OTHERS = 3.

  IF sy-subrc <> 0.
    MESSAGE 'Patient record locked by another user' TYPE 'E'.
  ENDIF.

  CLEAR gs_patient.

  gs_patient-patient_id        = gv_new_id.
  gs_patient-first_name        = p_fname.
  gs_patient-last_name         = p_lname.
  gs_patient-dob     = p_dob.
  gs_patient-registration_date = p_regdt.
  gs_patient-department_code   = p_mdept.
  gs_patient-blood_group       = p_bgrp.
  gs_patient-gender            = p_gen.
  gs_patient-phone             = p_phone.
  gs_patient-email_id          = p_email.
  gs_patient-city              = p_city.
  gs_patient-region_code       = p_reg.

  INSERT zbb_patient FROM gs_patient.

  IF sy-subrc = 0.

    COMMIT WORK.

    CALL FUNCTION 'DEQUEUE_EZBB_PATIENT'
      EXPORTING
        patient_id = gv_new_id.

    MESSAGE |Patient Registered Successfully. New Patient ID : { gv_new_id }|
    TYPE 'S'.

  ELSE.

    ROLLBACK WORK.

    CALL FUNCTION 'DEQUEUE_EZBB_PATIENT'
      EXPORTING
        patient_id = gv_new_id.

    MESSAGE 'Insert Failed' TYPE 'E'.

  ENDIF.

ENDFORM.
